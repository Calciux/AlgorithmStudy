# 高精度加法
```C++
int main()
{
    string a, b;
    cin >> a >> b;
    string ans;
    // 补前导零
    int len = max(a.size(), b.size());
    a = string(len - a.size(), '0') + a;
    b = string(len - b.size(), '0') + b;
    int carry = 0;
    for (int i = len - 1; i >= 0; i--)
    {
        int aa = a[i] - '0', ba = b[i] - '0';
        int out = aa + ba + carry;
        carry =out/10;
        out %= 10;
        char c = out + '0';
        ans = c + ans;
    }
    if (carry)
        cout << '1' + ans;
    else
        cout << ans;
    return 0;
}
```

```C++
// 首先判断数字大小
bool lessThan(string &a, string &b)
{
    if (a.size() != b.size())
        return a.size() < b.size();
    return a < b;
}
//去除前导零
string removeLeadingZ(string s){
    auto pos = s.find_first_not_of('0');
    if(pos==string::npos)
        return "0";
    s.erase(0,pos);
    return s;
}
// 高精度减法
string substract(string a, string b)
{
    bool negative = false;
    if (lessThan(a, b))
    {
        negative = true;
        swap(a, b);
    }
    // 补前导零
    int len = max(a.length(), b.length());
    a = string(len - a.size(), '0') + a;
    b = string(len - b.size(), '0') + b;

    string result = "";
    int borrow = 0;
    for (int i = len - 1; i >= 0; i--)
    {
        int x = (a[i] - '0') - borrow;
        int y = (b[i] - '0');
        // 需要借位
        if (x < y)
        {
            x += 10;
            borrow = 1;
        }
        else
        {
            borrow = 0;
        }
        result.push_back(x - y + '0');
        
    }
    reverse(result.begin(), result.end());
    // 去除前导零
    result = removeLeadingZ(result);
    if(negative&&result!="0")
    {
        result = '-' + result;
    }
    return result;
}

int main(){
    string a, b;
    cin >> a >> b;
    cout << substract(a, b) << endl;
    return 0;
}
```

```C++
string hpmulti(string a, string b)
{
    if (a == "0" || b == "0")
        return "0";
    int n = a.size();
    int m = b.size();
    vector<int> res(n + m, 0);
    reverse(a.begin(), a.end());
    reverse(b.begin(), b.end());
    // 中间结果
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < m; j++)
        {
            int x = (a[i] - '0') * (b[j] - '0');
            res[i + j] += x;
        }
    }
    // 处理进位
    for (int i = 0; i < n + m; i++)
    {
        res[i + 1] += res[i] / 10;
        res[i] %= 10;
    }
    // 转字符串
    string ans;
    for (int x : res)
    {
        ans.push_back(x + '0');
    }

}
```
