```C++
int LettertoNum(char c)
{
    if (c >= '0' && c <= '9')
        return c - '0';
    else if (c >= 'A' && c <= 'Z')
        return c - 'A' + 10;
}

int X_to_Dec(string s, int x)
{
    reverse(s.begin(), s.end());
    int n = s.size();
    int ans = 0;
    for (int i = 0; i < n; i++)
    {
        ans += LettertoNum(s[i]) * pow(x, i);
    }
    return ans;
}
```
